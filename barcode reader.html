function doPost(e) { try { const spreadsheetId = "1CAfKvnVcZLLuFn1yCWQBiJUmmGWOi8E6tbAI74a_xI0"; const sheetName = "MOUVEMENT BAIRS V3"; const payload = JSON.parse(e.postData.contents || "{}"); const spreadsheet = SpreadsheetApp.openById(spreadsheetId); const sheet = spreadsheet.getSheetByName(sheetName); if (!sheet) { throw new Error("Feuille introuvable : " + sheetName); } const timezone = spreadsheet.getSpreadsheetTimeZone() || Session.getScriptTimeZone(); const timestamp = Utilities.formatDate(new Date(), timezone, "dd/MM/yyyy HH:mm:ss"); const firstDataRow = 8; const nextRow = Math.max(sheet.getLastRow() + 1, firstDataRow); sheet.getRange(nextRow, 5, 1, 3).setValues([[ timestamp, payload.qrData || "", payload.location || "" ]]); return ContentService .createTextOutput(JSON.stringify({ success: true, message: "Ligne ajoutee dans Google Sheets." })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService .createTextOutput(JSON.stringify({ success: false, message: error.message })) .setMimeType(ContentService.MimeType.JSON); } }
